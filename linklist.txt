/*INSERTION*/
 Node *insertAtBegining(Node *head, int x) {
       Node *temp=new Node(x);
       if(head==NULL) return temp;
       
       temp->next=head;
       return temp;
    }
    
    
    
    Node *insertAtEnd(Node *head, int x)  {
      Node *temp=new Node(x);
      if(head==NULL) return temp;
      Node *curr=head;
      while(curr->next!=NULL)
      {
          curr=curr->next;
      }
        curr->next=temp;
        return head;   
       
    }
